#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>

struct node {
  struct node* left;
  struct node* right;
  int32_t val;
};

uint64_t task1(struct node *root)
{
  int32_t val;
  int32_t left;
  int32_t right;
  
  if (root != NULL) {
    val = root->val;
    left = task1(root->left);
    right = task1(root->right);
    return (uint64_t)(right + val * val + left);
  }
  return 0;
}
