struct node {
  struct node* left;
  struct node* right;
  char* str;
  uint64_t* res;
};

uint64_t* task2(struct node* root, char* target)

{
  int32_t diff;

  if (root != NULL) {
    diff = strcmp(target,root->str);
    if (diff == 0) {
      root = (uint64_t *)root->res;
    }
    else {
      if (diff < 0) {
        root = (uint64_t *)task2(root->left,target);
      }
      else {
        root = (uint64_t *)task2(root->right,target);
      }
    }
  }
  return root;
}
